GLM_models <- lapply(
  1:5,
  function(i) {
    myformula <- as.formula(paste0(
      "flow.2023 ~ total_pop_c_",
      i,
      " + wk_pop_c_",
      i,
      " + road_density"
    ))
    
    m <- glm(myformula,
             data = model_data_noNAs,
             family = poisson())
  }
)
